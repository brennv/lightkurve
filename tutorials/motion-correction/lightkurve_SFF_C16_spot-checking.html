def get_best_aper(hdu):
    '''Return the K2SFF best aperture for an input HDU'''
    ap_extension_lookup = {'CIRCULAR':22, 'PRF FIT':23}
    circ_or_prf = hdu[1].header['MASKTYPE']
    mask_index = hdu[1].header['MASKINDE']
    best_ap = hdu[ap_extension_lookup[circ_or_prf]].data[mask_index, :, :]
    return best_ap.astype(np.bool)

df_sff_metadata = pd.DataFrame()
i=0
for epic_id in tqdm(matched_ids):

    ## TPFs
    tpf_fn = '/Volumes/Truro/c16/ktwo{}-c16_lpd-targ.fits.gz'.format(epic_id)
    tpf = lightkurve.KeplerTargetPixelFile(tpf_fn)

    ## K2SFF
    k2sff_fn = '/Users/obsidian/Desktop/K2SFF/hlsp_k2sff_k2_lightcurve_{}-c16_kepler_v1_llc.fits'.format(epic_id)
    hdus = fits.open(k2sff_fn)
    hdu_num = 1
    meta = {key:hdus[hdu_num].data[key] for key in ['FRAW', 'ARCLENGTH', 'MOVING']}
    lc_k2sff = lightkurve.KeplerLightCurve(time=hdus[hdu_num].data['T'],
                                 flux=hdus[hdu_num].data['FCOR'],
                                 cadenceno=hdus[hdu_num].data['CADENCENO'],
                                 meta=meta)

    df_sff_metadata.at[i, 'EPIC_ID'] = epic_id
    df_sff_metadata.at[i, 'MASKTYPE'] = hdus[hdu_num].header['MASKTYPE']
    df_sff_metadata.at[i, 'MASKINDE'] = hdus[hdu_num].header['MASKINDE']

    ## Extract the TPF with the same aperture as K2SFF
    best_aper = get_best_aper(hdus)
    lc_raw = tpf.to_lightcurve(aperture_mask=best_aper)
    lc_match = lc_raw[np.in1d(lc_raw.cadenceno, lc_k2sff.cadenceno)].normalize()

    ## Compute the raw residual
    raw_residual = lc_match.flux - lc_k2sff.meta['FRAW']
    #bins = np.arange(-0.1, 0.1, 0.005)
    #plt.hist(raw_residual, bins=bins, alpha=0.01, color='r');
    #plt.axvline(0, linestyle='dashed', c='k')
    df_sff_metadata.at[i, 'raw_resid'] = np.nanstd(raw_residual)
    #plt.show()


    ## Compute SFF
    try:
        lc_corrected = sff.correct(lc_match.time, lc_match.flux, lc_match.centroid_col, lc_match.centroid_row,
                           niters=2, windows=19, polyorder=5, restore_trend=True)


        sff_residual = lc_corrected.flux - lc_k2sff.flux
        #bins = np.arange(-1, 1, 0.05)
        #plt.hist(sff_residual, bins=bins, alpha=0.01, color='r');
        #plt.axvline(0, linestyle='dashed', c='k')
        df_sff_metadata.at[i, 'sff_resid'] = np.nanstd(sff_residual)

        #plt.scatter(lc_k2sff.time, sff_residual, s=1, alpha=0.01, color='r', marker='o')
        plt.scatter(lc_k2sff.time, lc_k2sff.flux, s=1, marker='o')
        plt.scatter(lc_k2sff.time, lc_corrected.flux, s=1, marker='o')
        plt.show()
    except:
        print("{} had a problem".format(i))

    #plt.show()
    i+=1

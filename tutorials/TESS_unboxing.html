Now we can download the data with lightkurve made by the Kepler/K2 Guest Observer Office.

In [6]:

from lightkurve import TessTargetPixelFile

In [7]:

tic_id = 25375553

In [8]:

tpf_path = 'https://archive.stsci.edu/hlsps/tess-data-alerts/'\
                'hlsp_tess-data-alerts_tess_phot_{:011d}-s01_tess_v1_tp.fits'.format(tic_id)

In [9]:

tpf = TessTargetPixelFile(tpf_path)

In [10]:

tpf.plot();

In [11]:

lc = tpf.to_lightcurve()

In [12]:

lc.plot()
